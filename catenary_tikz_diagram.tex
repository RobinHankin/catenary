\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}
\begin{document}


\begin{tikzpicture}[scale=1.3,
    force/.style = {
        -{Stealth[length=3.5mm, width=1.2mm, inset=0.8pt]}, 
        thick
    },
    label/.style = {midway, fill=white, inner sep=2pt, font=\small}
]

\draw[thick, blue]
  plot[domain=0:2, samples=100]
    ({\x}, {cosh(\x) - 1});

\fill (0,0) circle (1.2pt);
\node[below] at (0,0){$(0,1)$};

\fill (2,{cosh(2)-1}) circle (1.2pt);
\node[left] at (2,{cosh(2)-1}) {$(x,\cosh(x))$};

\draw[force] (0,0) -- ++(-1,0)
    node[label] {$T_0$};

\draw[force] (2,{cosh(2)-1}) -- ++(1,0)
    node[label] {$T_x$};
\draw[force] (2,{cosh(2)-1}) -- ++(0,1.2)
    node[label] {$T_y$};

\draw[force]
  (1.24, {cosh(1.24)-1}) -- ++(0,-1)
    node[right] {mg};

\draw[thin,dashed]
  (1.24, {cosh(1.24)-1}) -- ++(0, 0.5)
    node[right] {};

\end{tikzpicture}
